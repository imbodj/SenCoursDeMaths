\begin{exercice}[Décomposition]

\begin{enumerate}
\item  $ h(x)= \sqrt{7x+1} $ 


\item  $ f(x)= \paren{5x+6}^{2} $  


\item  $ h(x)= \paren{x-2}^{2} -4 $ 



\item $ h(x)= \dfrac{\sqrt{x}+1}{1-2\sqrt{x}} $ 
\end{enumerate}

\end{exercice}
