% Listes :

% Configuration des listes enumerate
\setlist[enumerate,1]{label=(\roman*), leftmargin=*, itemsep=0.2em}
\setlist[enumerate,2]{label=(\alph*), leftmargin=*, itemsep=0.1em}
\setlist[enumerate,3]{label=\arabic*., leftmargin=*, itemsep=0.1em}
\setlist[enumerate,4]{label=(\arabic*), leftmargin=*, itemsep=0.1em}

% Configuration des listes itemize
\setlist[itemize,1]{label=\textbullet, leftmargin=*, itemsep=0.2em}
\setlist[itemize,2]{label=\textendash, leftmargin=*, itemsep=0.1em}
\setlist[itemize,3]{label=\textasteriskcentered, leftmargin=*, itemsep=0.1em}
\setlist[itemize,4]{label=\textperiodcentered, leftmargin=*, itemsep=0.1em}

% Configuration des listes description
\setlist[description]{leftmargin=\parindent, itemsep=0.2em, font=\bfseries}

% Configuration globale pour toutes les listes
\setlist{
    topsep=0.3em,        % Espace avant la liste
    partopsep=0.2em,     % Espace supplémentaire si la liste commence un nouveau paragraphe
    parsep=0.1em,        % Espace entre les paragraphes dans un même item
    listparindent=\parindent  % Indentation des paragraphes dans les items
}


% Commandes :

\providecommand{\lesson}[3]{%
	\begin{docname}%
		\section*{\textbf{#2} #3}%
	\end{docname}%
	\begin{doctitle}%
		#3%
	\end{doctitle}%
	\begin{doccategories}%
		#1%
	\end{doccategories}%
}

\providecommand{\development}[3]{%
	\begin{docname}%
		\section*{#3}%
	\end{docname}%
	\begin{doctitle}%
		#3%
	\end{doctitle}%
	\begin{doccategories}%
		#1%
	\end{doccategories}%
}

\providecommand{\sheet}[3]{\development{#1}{#2}{#3}}

\providecommand{\ranking}[1]{%
	\begin{docname}%
		\section*{Terminale #1}%
	\end{docname}%
}

\providecommand{\summary}[1]{%
	\begin{docsummary}%
		#1%
	\end{docsummary}%
	\newpar%
}

\providecommand{\reference}[2][]{%
	\begin{bookref}%
		\begin{bookrefshort}#1\end{bookrefshort}%
		\begin{bookrefpage}p. #2\end{bookrefpage}%
	\end{bookref}%
}

\providecommand{\dev}[1]{%
	\begin{devlink}#1\end{devlink}%
}

\providecommand{\newpar}{


}
